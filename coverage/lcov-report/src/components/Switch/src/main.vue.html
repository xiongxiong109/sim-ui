<template>
  <div
    class="sm-switch__wrapper"
    :class="{'is-open': this.value}"
    @click="evtClick"
  ></div>
</template>
<script>
// 开关选择按钮
export default {
  name: 'sm-switch',
  props: {
    value: { // v-model双向绑定值
      type: Boolean,
      default: false
    },
    isDisabled: { // 是否禁止点击
      type: Boolean,
      default: false
    }
  },
  methods: {
    evtClick() {
      !this.isDisabled && this.$emit('input', !this.value);
    }
  }
}
</script>
